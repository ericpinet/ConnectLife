012package com.digi.xbee.api.exceptions;
013
014/**
015 * This exception will be thrown when trying to perform any operation with 
016 * the XBee device and its communication interface is closed.
017 * 
018 * @see RuntimeException
019 */
020public class InterfaceNotOpenException extends RuntimeException {
021
022        // Constants
023        private static final long serialVersionUID = 1L;
024        private static final String DEFAULT_MESSAGE = "The connection interface is not open.";
025
026        /**
027         * Creates an {@code InterfaceNotOpenException} with 
028         * {@value #DEFAULT_MESSAGE} as its error detail message.
029         */
030        public InterfaceNotOpenException() {
031                super(DEFAULT_MESSAGE);
032        }
033        
034        /**
035         * Creates an {@code InterfaceNotOpenException} with the specified 
036         * message.
037         * 
038         * @param message The associated message.
039         */
040        public InterfaceNotOpenException(String message) {
041                super(message);
042        }
043        
044        /**
045         * Creates an {@code InterfaceNotOpenException} with the specified message
046         * and cause.
047         * 
048         * @param message The associated message.
049         * @param cause The cause of this exception.
050         * 
051         * @see Throwable
052         */
053        public InterfaceNotOpenException(String message, Throwable cause) {
054                super(message, cause);
055        }
056}
