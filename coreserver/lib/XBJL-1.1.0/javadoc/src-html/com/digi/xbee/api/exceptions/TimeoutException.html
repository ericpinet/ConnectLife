012package com.digi.xbee.api.exceptions;
013
014/**
015 * This exception will be thrown when performing synchronous operations and the
016 * configured time expires.
017 * 
018 * @see CommunicationException
019 */
020public class TimeoutException extends CommunicationException {
021
022        // Constants
023        private static final long serialVersionUID = 1L;
024        private static final String DEFAULT_MESSAGE = "There was a timeout while executing the requested operation.";
025
026        /**
027         * Creates a {@code TimeoutException} with {@value #DEFAULT_MESSAGE} as its 
028         * error detail message.
029         */
030        public TimeoutException() {
031                super(DEFAULT_MESSAGE);
032        }
033        
034        /**
035         * Creates a {@code TimeoutException} with the specified message.
036         * 
037         * @param message The associated message.
038         */
039        public TimeoutException(String message) {
040                super(message);
041        }
042        
043        /**
044         * Creates a {@code TimeoutException} with the specified message and cause.
045         * 
046         * @param message The associated message.
047         * @param cause The cause of this exception.
048         * 
049         * @see Throwable
050         */
051        public TimeoutException(String message, Throwable cause) {
052                super(message, cause);
053        }
054}
